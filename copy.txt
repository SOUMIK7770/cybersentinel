# ── API: Gemini ────────────────────────────────────────────────────────────────
def call_gemini(prompt: str, retries: int = 3) -> str:
    if not GEMINI_API_KEY:
        return "__ERROR__No Gemini API key configured. Please add GEMINI_API_KEY to your .env file."
    
    delays = [2, 4, 6]
    headers = {"Content-Type": "application/json"}
    payload = {
        "contents": [{"parts": [{"text": prompt}]}],
        "generationConfig": {
            "temperature": 0.4,
            "maxOutputTokens": 1500,
            "topP": 0.9,
        }
    }
    url = f"{GEMINI_ENDPOINT}?key={GEMINI_API_KEY}"

    for attempt in range(retries):
        try:
            resp = requests.post(url, headers=headers, json=payload, timeout=30)
            if resp.status_code == 200:
                data = resp.json()
                return data["candidates"][0]["content"]["parts"][0]["text"]
            elif resp.status_code == 429:
                if attempt < retries - 1:
                    time.sleep(delays[attempt])
                    continue
            elif resp.status_code in (500, 503):
                if attempt < retries - 1:
                    time.sleep(delays[attempt])
                    continue
            return f"__ERROR__API returned {resp.status_code}. Please try again."
        except requests.exceptions.Timeout:
            if attempt < retries - 1:
                time.sleep(delays[attempt])
                continue
            return "__ERROR__Request timed out. Please try again."
        except Exception as e:
            if attempt < retries - 1:
                time.sleep(delays[attempt])
                continue
            return "__ERROR__Connection error. Please try again."
    return "__ERROR__Analysis temporarily unavailable. Please try again."

# ── API: VirusTotal ────────────────────────────────────────────────────────────
def query_virustotal(url_to_scan: str) -> dict:
    if not VIRUSTOTAL_API_KEY:
        return {"error": "No VirusTotal API key configured. Please add VIRUSTOTAL_API_KEY to your .env file."}
    
    try:
        url_id = base64.urlsafe_b64encode(url_to_scan.encode()).decode().rstrip("=")
        headers = {"x-apikey": VIRUSTOTAL_API_KEY}
        
        resp = requests.get(
            f"https://www.virustotal.com/api/v3/urls/{url_id}",
            headers=headers, timeout=20
        )

        if resp.status_code == 200:
            stats = resp.json()["data"]["attributes"]["last_analysis_stats"]
            return {
                "malicious":  stats.get("malicious",  0),
                "suspicious": stats.get("suspicious", 0),
                "harmless":   stats.get("harmless",   0),
                "undetected": stats.get("undetected", 0),
            }

        # URL not in VT db yet — submit then poll
        if resp.status_code == 404:
            sub = requests.post(
                "https://www.virustotal.com/api/v3/urls",
                headers=headers,
                data={"url": url_to_scan},
                timeout=20,
            )
            if sub.status_code != 200:
                return {"error": f"VirusTotal submission failed ({sub.status_code})."}
            
            analysis_id = sub.json()["data"]["id"]
            for _ in range(10):
                time.sleep(3)
                poll = requests.get(
                    f"https://www.virustotal.com/api/v3/analyses/{analysis_id}",
                    headers=headers, timeout=20
                )
                if poll.status_code == 200:
                    pd = poll.json()["data"]
                    if pd["attributes"]["status"] == "completed":
                        s = pd["attributes"]["stats"]
                        return {
                            "malicious":  s.get("malicious",  0),
                            "suspicious": s.get("suspicious", 0),
                            "harmless":   s.get("harmless",   0),
                            "undetected": s.get("undetected", 0),
                        }
            return {"error": "Analysis timed out. Please try again."}

        return {"error": f"VirusTotal error ({resp.status_code})."}

    except requests.exceptions.Timeout:
        return {"error": "VirusTotal request timed out."}
    except Exception as e:
        return {"error": "Could not reach VirusTotal. Check your connection."}

def analyze_url(url: str) -> dict:
    cache_key = f"url::{url}"
    if cache_key in st.session_state.cache:
        return st.session_state.cache[cache_key]

    vt = query_virustotal(url)
    if "error" in vt:
        result = {"error": vt["error"]}
        st.session_state.cache[cache_key] = result
        return result

    malicious  = vt["malicious"]
    suspicious = vt["suspicious"]
    harmless   = vt["harmless"]
    undetected = vt["undetected"]
    total      = malicious + suspicious + harmless + undetected or 1

    if malicious >= 3:
        level = "HIGH"
    elif malicious >= 1 or suspicious >= 3:
        level = "MEDIUM"
    elif suspicious >= 1:
        level = "MEDIUM" if suspicious >= 2 else "LOW"
    else:
        level = "LOW"

    prompt = f"""You are a cybersecurity expert. Analyze this URL threat report and produce a structured assessment.

URL: {url}
VirusTotal Results:
- Malicious detections: {malicious}
- Suspicious detections: {suspicious}
- Harmless verdicts: {harmless}
- Undetected/Unknown: {undetected}
- Overall threat level: {level}

Respond in this EXACT JSON format (no markdown, no explanation outside JSON):
{{
  "summary": "2-3 sentence human-friendly explanation of what this URL's risk profile means",
  "risks": ["risk point 1", "risk point 2", "risk point 3"],
  "recommendations": ["action 1", "action 2", "action 3"]
}}"""

    raw = call_gemini(prompt)
    if raw.startswith("__ERROR__"):
        result = {"error": raw.replace("__ERROR__", "")}
        st.session_state.cache[cache_key] = result
        return result

    try:
        clean = re.sub(r"```(?:json)?", "", raw).strip().rstrip("`").strip()
        parsed = json.loads(clean)
    except Exception:
        parsed = {
            "summary": raw[:300],
            "risks": ["Could not parse structured risks."],
            "recommendations": ["Review the raw analysis above."]
        }

    result = {
        "level":            level,
        "malicious":        malicious,
        "suspicious":       suspicious,
        "harmless":         harmless,
        "total":            total,
        "summary":          parsed.get("summary", ""),
        "risks":            parsed.get("risks", []),
        "recommendations":  parsed.get("recommendations", []),
        "target":           url,
        "type":             "URL",
    }
    st.session_state.cache[cache_key] = result
    return result
